// stack.c
// 栈的简单操作--DFS 深度优先搜索
// Created By CodeMax At 2022/07/06/ 19:33
// FILO
#include <stdio.h>

char stack[512];
int top = 0;

void push(char c) { stack[top++] = c; }

char pop(void) { return stack[--top]; }

int is_empty(void) { return top == 0; }

int main(void)
{
    push('a');
    push('b');
    push('c');

    while (!is_empty()) {
        putchar(pop());
        putchar(' ');
    }
    
    return 0;
}
// c b a

指针

  • 指针和const
1
2

const int *p;
int const *p;

指的是：p指向的内存单元不能改写，即(*p)++是不允许的, p++是可以的，说明p可指向其他内存单元。

1

int *const p;

指的是：p是一个指向int的const指针, (*p)可以改写， 但是p不能改写

1

int const * const p;

指的是： p和*p都能被改写

多用const对指针进显式限制，同时能够检查唯一性。
